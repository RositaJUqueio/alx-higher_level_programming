import unittest
from 0-add_integer import add_integer

class AddIntegerTestCase(unittest.TestCase):

    def setUp(self):
        # Arrange test data
        self.a = 2.3
        self.b = 8

    def test_add_integer(self):
        # Act
        result = add_integer(self.a, self.b)

        # Assert
        expected = 10
        self.assertEqual(result, expected)

if __name__ == '__main__':
    unittest.main()
